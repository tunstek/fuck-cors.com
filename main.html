    Here's an example:

    
$.ajax({
  url: "http://fuck-cors.com/?url=http://ip.jsontest.com/",
  dataType: "text",
  success: function (data) {
    console.log(data)
  },
  error: function (xhr, ajaxOptions, thrownError) {
    console.log(xhr.status + thrownError);
  }
});
